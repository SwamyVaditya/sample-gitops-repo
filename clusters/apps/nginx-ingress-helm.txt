Install NGINX Ingress Controller on apps-cluster

helm repo add ingress-nginx https://kubernetes.github.io/ingress-nginx
helm repo update
helm install ingress-nginx ingress-nginx/ingress-nginx --namespace ingress-nginx --create-namespace


Verify:

kubectl get pods -n ingress-nginx  # Controller Running (1/1)
kubectl get svc -n ingress-nginx  # LoadBalancer, ports 80/TCP,443/TCP
kubectl get ingressclass  # nginx listed
